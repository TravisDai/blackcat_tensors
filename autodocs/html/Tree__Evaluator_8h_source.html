16 /*
17  *
18  * This file defines the endpoints for the Tree_Evaluator. The Tree Evaluator
19  * iterates through the tree and attempts use the the left-hand value as a cache (if possible)
20  * and to reduce the need for temporaries when applicable.
21  *
22  * After utilizing the left-hand value as cache, the tree is iterated through again, greedy-evaluating
23  * any function calls when possible. A function call is generally any BLAS call.
24  *
25  * Example: (assume matrices)
26  * y += a * b + c * d
27  *
28  * Naively, this expression may generate 2 temporaries, one for each matrix multiplication.
29  * However, a more efficient way to evaluate this equation would be to make 2 gemm calls,
30  * gemm(y,a,b) and gemm(y, c, d).
31  *
32  * This expression reordering works with more complex calls, such as....
33  * y = abs(a * b + c * d).
34  *
35  * Here we can apply... (the second gemm call updateing alpha to 1)
36  * gemm(y,a,b), gemm(y,c,d) followed by evaluating y := abs(y).
37  *
38  */

55 template<class AssignmentOp, class Left, class Right, class SystemTag>
56 void greedy_eval(Left left, Right right, BC::Stream<SystemTag> stream) {
57  auto temporaries_evaluated_expression = optimizer<Right>::template temporary_injection(right, stream);
58  auto assignment_expression = make_bin_expr<AssignmentOp>(left, temporaries_evaluated_expression);
59  detail::nd_evaluate(assignment_expression, stream);
60  optimizer<decltype(temporaries_evaluated_expression)>::deallocate_temporaries(temporaries_evaluated_expression, stream);
61 }

78  auto output = injector<lv, alpha_mod, beta_mod>(expression.left);
79  auto right = optimizer<rv>::linear_evaluation(expression.right, output, stream);
80 
81  if /*constexpr*/ (!entirely_blas_expression)
82  detail::greedy_eval<Op>(expression.left, right, stream);
83 }
