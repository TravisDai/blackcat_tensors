112  * If left/right is/are transposed, calling 'dimensions(0), rows(), cols()'
113  * will return the non-transposed dimensions/rows/cols. Ergo- you should use the original parameters
114  * to access the shape of the returned value.
115  */
116

118  static constexpr bool rv_scalar = blas_expression_traits<Rv>::is_scalar_multiplied;
119  using value_type = typename Lv::value_type;
120 
121  auto alpha_lv = blas_expression_traits<Lv>::get_scalar(left);
122  auto alpha_rv = blas_expression_traits<Rv>::get_scalar(right);
123 
124  auto alpha_ = calculate_alpha<value_type, alpha_mod, lv_scalar, rv_scalar>(stream, alpha_lv, alpha_rv);
125  auto beta_ = make_constexpr_scalar<typename expression_traits<decltype(alpha_)>::allocation_tag, beta_mod, value_type>();//blas_impl::template scalar_constant<value_type, beta_mod>();

127  auto left_ = greedy_evaluate(blas_expression_traits<Lv>::remove_blas_modifiers(left), stream);

129

131  using right_t = std::decay_t<decltype(right_)>;
132  using alpha_t = std::decay_t<decltype(alpha_)>;
133  using beta_t = std::decay_t<decltype(beta_)>;

135  return contents<
