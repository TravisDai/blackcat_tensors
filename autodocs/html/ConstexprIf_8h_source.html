37 //-------------------------- constexpr if -----------------------//
38 /*
39  * C++ 11/14 version of constexpr if. (Required because NVCC doesn't support C++17)
40  *
41  * Accepts a constexpr bool template argument, and one or two functors (that take no arguments)
42  * if true, calls and returns the first functor, else the second.
43  *
44  * Example:
45  *
46  * int main() {
47  * static constexpr bool value = false;
48  * return BC::meta:constexpr_if<false>(
49  * []() { std::cout << " constexpr_boolean is true " << std::endl; return true;},
50  * []() { std::cout << " constexpr_boolean is false " << std::endl; return false; }
51  * );
52  *}
53  * output: constexpr_boolean is false
54  */
