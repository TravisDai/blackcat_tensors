124  template<class core, int a, int b, class Stream>
125  static auto linear_evaluation(Binary_Expression<op, lv, rv>& branch, injector<core, a, b> tensor, Stream alloc) {
126  return
127  BC::meta::constexpr_if<entirely_blas_expr>(
128  [&](){
129  optimizer<lv>::linear_evaluation(branch.left, tensor, alloc);
130  optimizer<rv>::linear_evaluation(branch.right, update_injection<op, true>(tensor), alloc);
131  return tensor.data();
132  },
133  BC::meta::constexpr_else_if<optimizer<lv>::entirely_blas_expr>(
134  [&]() {
135  return BC::meta::constexpr_ternary<std::is_same<op, oper::Sub>::value>(
136  [&]() {
137  /*auto left = */ optimizer<lv>::linear_evaluation(branch.left, tensor, alloc);
138  auto right = optimizer<rv>::linear_evaluation(branch.right, update_injection<op, true>(tensor), alloc);
139  return make_un_expr<oper::negation>(right);
140 
141  },
142  [&]() {
143  /*auto left = */ optimizer<lv>::linear_evaluation(branch.left, tensor, alloc);
144  auto right = optimizer<rv>::linear_evaluation(branch.right, update_injection<op, true>(tensor), alloc);
145  return right;
146  }
147  );
148  },
149  BC::meta::constexpr_else_if<optimizer<rv>::entirely_blas_expr>(
150  [&]() {
151  auto left = optimizer<lv>::linear_evaluation(branch.left, tensor, alloc);
152  /*auto right = */ optimizer<rv>::linear_evaluation(branch.right, update_injection<op, b != 0>(tensor), alloc);
153  return left;
154  },
155  BC::meta::constexpr_else_if<optimizer<rv>::nested_blas_expr>(
156  [&]() {
157  auto right = optimizer<rv>::injection(branch.right, tensor, alloc);
158  return make_bin_expr<op>(branch.left, right);

163  auto left = optimizer<lv>::linear_evaluation(branch.left, tensor, alloc);
164  auto right = optimizer<rv>::linear_evaluation(branch.right, update_injection<op, left_evaluated>(tensor), alloc);
