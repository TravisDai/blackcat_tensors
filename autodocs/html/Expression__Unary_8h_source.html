102  struct dx_defined {
103  template<class T> BCINLINE
104  static auto impl(const T& array, const Operation& op, BC::size_t index) {
105  // ’(f(g(x))) == f’(g(x))g’(x)
106  auto gx_and_dxgx = array.dx(index); // [g(x), g’(x)]
107  auto gx = gx_and_dxgx.first; // g(x)
108  auto dx_gx = gx_and_dxgx.second; // g’(x)
109 
110  auto f_gx = op(gx);
111  auto dx_f_gx = op.dx(gx) * dx_gx;
112 
113  return BC::meta::make_pair(f_gx, dx_f_gx);
114  }
115  };
